// Ejemplo simplificado de cómo se lee el peso en el Frontend
async function conectarBascula() {
    try {
        // 1. El navegador pide al usuario que seleccione el puerto (COM3)
        const port = await navigator.serial.requestPort();

        // 2. Abrimos el puerto con la configuración estándar de Toledo
        await port.open({ baudRate: 9600, dataBits: 8, parity: 'none', stopBits: 1 });

        const reader = port.readable.getReader();
        const decoder = new TextDecoder();
        let pesoCrudo = '';

        // 3. Bucle infinito leyendo lo que envía la báscula
        while (true) {
            const { value, done } = await reader.read();
            if (done) break;

            pesoCrudo += decoder.decode(value);

            // 4. Limpiamos la data (Toledo envía caracteres invisibles como STX y ETX)
            // Extraemos solo los números
            let pesoLimpio = extraerPesoToledo(pesoCrudo); 

            if (pesoLimpio) {
                // 5. Inyectamos el peso en tu formulario (Ej: 150.00)
                document.getElementById('inputPesadaEntrada').value = pesoLimpio;
                pesoCrudo = ''; // Reiniciamos para la siguiente lectura
            }
        }
    } catch (error) {
        console.error("Error al conectar la báscula:", error);
    }
}

Python
import serial
import threading
import time
import re
from flask import Flask, jsonify
from flask_cors import CORS

# 1. Configuración del Servidor Local
app = Flask(__name__)
CORS(app)  # Permite que tu sistema web (ej. https://globalcafe.com) consulte este localhost

# Variable global que almacenará el último peso leído
peso_actual = "0.00"

def limpiar_datos_toledo(raw_data):
    """
    Las básculas Toledo envían caracteres de control invisibles (STX, ETX, CR, LF).
    Esta función limpia la basura y extrae solo el número decimal.
    """
    try:
        texto = raw_data.decode('ascii', errors='ignore')
        # Busca el patrón de números enteros o con decimales
        numeros = re.findall(r'\d+\.\d+|\d+', texto)
        if numeros:
            return numeros[0]
    except Exception as e:
        pass
    return None

def escuchar_bascula():
    """
    Esta función correrá en segundo plano (hilo) 24/7 leyendo el puerto COM.
    """
    global peso_actual
    puerto_com = 'COM3'  # <-- CAMBIA ESTO AL PUERTO REAL DE TU PC
    
    while True:
        try:
            # Configuración estándar para Toledo TC420
            with serial.Serial(puerto_com, baudrate=9600, bytesize=8, parity='N', stopbits=1, timeout=1) as puerto:
                print(f"✅ Conectado exitosamente a la báscula en {puerto_com}")
                
                while True:
                    if puerto.in_waiting > 0:
                        # Leemos la línea que envía la báscula
                        raw_data = puerto.readline() 
                        peso_limpio = limpiar_datos_toledo(raw_data)
                        
                        if peso_limpio:
                            peso_actual = peso_limpio
                            # console.log opcional para ver en vivo
                            # print(f"Peso leído: {peso_actual} kg") 
                            
                    time.sleep(0.05) # Pequeña pausa para no saturar el CPU
                    
        except serial.SerialException:
            print(f"⚠️ Error: No se pudo conectar a {puerto_com}. Reintentando en 5 segundos...")
            time.sleep(5)
        except Exception as e:
            print(f"⚠️ Error inesperado: {e}")
            time.sleep(5)

# 2. El Endpoint que consultará tu Frontend en la Nube
@app.route('/peso', methods=['GET'])
def obtener_peso():
    # Devuelve el último peso capturado en formato JSON
    return jsonify({"peso": peso_actual})

if __name__ == '__main__':
    print("Iniciando el Agente Local de Báscula PSS GLOBAL...")
    
    # Iniciar la lectura de la báscula en un hilo (background)
    hilo_bascula = threading.Thread(target=escuchar_bascula, daemon=True)
    hilo_bascula.start()
    
    # Levantar el micro-servidor web en el puerto 4000
    # host='127.0.0.1' asegura que solo la propia computadora pueda consultarlo
    app.run(host='127.0.0.1', port=4000, debug=False)

1. ¿Habrá problemas con los puertos abiertos o procesos colgados?
Con el primer código que te pasé (que usa un hilo continuo o "background thread"), SÍ, existe un riesgo a largo plazo.
En entornos reales (plantas, casetas), ocurren dos cosas que "cuelgan" los puertos COM:

Micro-cortes de energía: La báscula parpadea, el cable pierde voltaje 1 milisegundo y el puerto COM de Windows se bloquea.

Suspensión Selectiva de USB: Windows, para ahorrar energía, "duerme" los puertos USB/Serial si no los usas por unas horas. Cuando el script intenta leer de nuevo, el proceso se congela.

2. Tu idea: Abrir y cerrar el puerto bajo demanda
Lo que acabas de sugerir ("abres y cierras los puertos cuando captura peso") es la Estrategia Definitiva y Más Segura.

En lugar de tener al programa escuchando 24/7 y gastando RAM, hacemos que el .exe sea "perezoso": solo abre el puerto cuando el usuario presiona el botón, lee el peso al instante, y cierra el puerto inmediatamente.

Aquí tienes el código Python refactorizado y optimizado con tu idea. Es mucho más corto, no usa hilos (threads) y jamás se va a quedar colgado:

Python
import serial
import time
import re
from flask import Flask, jsonify
from flask_cors import CORS

app = Flask(__name__)
CORS(app)

PUERTO_COM = 'COM3' # <-- Cambiar al tuyo

def limpiar_datos_toledo(raw_data):
    try:
        texto = raw_data.decode('ascii', errors='ignore')
        numeros = re.findall(r'\d+\.\d+|\d+', texto)
        if numeros:
            return numeros[0]
    except Exception:
        pass
    return None

# Endpoint que abre y cierra bajo demanda
@app.route('/peso', methods=['GET'])
def obtener_peso():
    try:
        # 1. ABRIMOS EL PUERTO (El "with" garantiza que se cierre automáticamente al terminar)
        # timeout=2 evita que se quede colgado si la báscula está apagada
        with serial.Serial(PUERTO_COM, baudrate=9600, bytesize=8, parity='N', stopbits=1, timeout=2) as puerto:
            
            # 2. LIMPIAR BASURA VIEJA: Borramos lo que haya quedado atascado en el cable
            puerto.reset_input_buffer()
            
            # 3. LEER: Intentamos leer hasta 3 líneas seguidas para asegurar un dato fresco
            for _ in range(3):
                raw_data = puerto.readline()
                peso = limpiar_datos_toledo(raw_data)
                
                if peso:
                    # 4. CIERRE Y RESPUESTA EXITOSA (El puerto se cierra solo al salir del "with")
                    return jsonify({"peso": peso, "estado": "exito"})
                    
            # Si leyó 3 veces y era pura basura:
            return jsonify({"peso": "0.00", "estado": "error", "mensaje": "Datos ilegibles de la báscula"}), 500

    except serial.SerialException as e:
        # El puerto se cerró solo por el error. No hay procesos colgados.
        return jsonify({"peso": "0.00", "estado": "error", "mensaje": "Báscula desconectada o puerto ocupado"}), 500
    except Exception as e:
        return jsonify({"peso": "0.00", "estado": "error", "mensaje": str(e)}), 500

if __name__ == '__main__':
    print("Agente de Báscula (Modo Bajo Demanda) Iniciado...")
    # Corre el servidor sin hilos extras
    app.run(host='127.0.0.1', port=4000, debug=False)
🧠 ¿Por qué este código es "A Prueba de Balas"?
Uso del bloque with serial.Serial(...): En Python, la palabra reservada with es un "Gestor de Contexto". Garantiza al 100% que el puerto se cierre automáticamente, incluso si hay un error, si se desconecta el cable, o si la báscula explota. Jamás quedará el puerto COM tomado o colgado.

puerto.reset_input_buffer(): Este es el secreto industrial. Borra la memoria del puerto. Así evitas que a las 3:00 PM el operario lea un peso que se quedó atascado en el cable desde las 10:00 AM.

timeout=2: Si la báscula está apagada, el proceso no se queda esperando al infinito. A los 2 segundos se rinde, cierra el puerto, y le avisa a tu Frontend que hay un error.
